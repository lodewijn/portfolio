x <- seq(0, 10, length.out = 100)
y <- exp(x / 3)

ggplot(data.frame(x, y), aes(x, y)) +
  geom_line(color = "red", size = 1.5) +
  labs(
    x = "My love for Ko and Piet",
    y = "How hungry they are \n / How cute they look to get food",
    title = "Love versus hunger"
  ) +
  theme_minimal(base_size = 16)
